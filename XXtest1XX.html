# .github/workflows/import-hostaway.yml # Ejecutar manualmente desde GitHub Actions → Run workflow # No necesita Node.js local ni permisos especiales name: Importar Hostaway → Caspio on: workflow_dispatch: inputs: hours: description: 'Horas hacia atrás a importar (ej: 72 = 3 días)' required: true default: '72' batch: description: 'Registros por llamada (max 15 para Cloudflare)' required: true default: '15' jobs: import: runs-on: ubuntu-latest timeout-minutes: 30 steps: - name: Importar reservas Hostaway → Caspio run: | HOURS="${{ github.event.inputs.hours }}" BATCH="${{ github.event.inputs.batch }}" WORKER_URL="https://importar-host-a-caspio-ultimas-3horas.jordi-89b.workers.dev/import" CALL=0 TOTAL_INSERTED=0 TOTAL_UPDATED=0 TOTAL_SKIPPED=0 TOTAL_ERRORS=0 URL="${WORKER_URL}?hours=${HOURS}&batch=${BATCH}&offset=0" echo "🚀 Iniciando importación — últimas ${HOURS}h" echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━" while [ -n "$URL" ]; do CALL=$((CALL + 1)) echo "" echo "📦 Llamada #${CALL} → ${URL}" RESPONSE=$(curl -s --max-time 30 "$URL") if [ $? -ne 0 ]; then echo "❌ Error de red en llamada #${CALL}" exit 1 fi echo " Respuesta: $RESPONSE" STATUS=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('status',''))") INSERTED=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('inserted',0))") UPDATED=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('updated',0))") SKIPPED=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('skipped',0))") ERRORS=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('errors',0))") HAS_MORE=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('hasMore',False))") NEXT_CALL=$(echo "$RESPONSE" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('nextCall',''))") if [ "$STATUS" != "ok" ]; then echo "❌ Worker devolvió error: $RESPONSE" exit 1 fi TOTAL_INSERTED=$((TOTAL_INSERTED + INSERTED)) TOTAL_UPDATED=$((TOTAL_UPDATED + UPDATED)) TOTAL_SKIPPED=$((TOTAL_SKIPPED + SKIPPED)) TOTAL_ERRORS=$((TOTAL_ERRORS + ERRORS)) echo " ✅ inserted:${INSERTED} updated:${UPDATED} skipped:${SKIPPED} errors:${ERRORS}" if [ "$HAS_MORE" != "True" ]; then echo "" echo "✅ No hay más registros — importación completada" break fi URL="$NEXT_CALL" # Pausa 2s para respetar rate limit Hostaway sleep 2 done echo "" echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━" echo "📊 RESUMEN FINAL (${CALL} llamadas):" echo " ✅ Insertados : ${TOTAL_INSERTED}" echo " 🔄 Actualizados: ${TOTAL_UPDATED}" echo " ⏭️ Saltados : ${TOTAL_SKIPPED}" echo " ❌ Errores : ${TOTAL_ERRORS}"
